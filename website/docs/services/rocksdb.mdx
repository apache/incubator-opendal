```rust
use anyhow::Result;
use opendal::Operator;
use opendal::Scheme;
use std::collections::HashMap;

#[tokio::main]
async fn main() -> Result<()> {
    let mut map = HashMap::new();
    map.insert("datadir".to_string(), "/tmp/opendal/rocksdb".to_string());

    let op: Operator = Operator::via_map(Scheme::Rocksdb, map)?;
    Ok(())
}
```
