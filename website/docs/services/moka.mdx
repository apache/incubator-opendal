use anyhow::Result;
use opendal::Operator;
use opendal::Scheme;
use std::collections::HashMap;

#[tokio::main]
async fn main() -> Result<()> {
    let mut map = HashMap::new();
    map.insert("name".to_string(), "your_cache_name".to_string());

    let op: Operator = Operator::via_map(Scheme::Moka, map)?;
    Ok(())
}
