use anyhow::Result;
use opendal::Operator;
use opendal::Scheme;
use std::collections::HashMap;

#[tokio::main]
async fn main() -> Result<()> {
    let mut map = HashMap::new();
    map.insert("endpoint".to_string(), "127.0.0.1".to_string());

    let op: Operator = Operator::via_map(Scheme::Http, map)?;
    Ok(())
}
