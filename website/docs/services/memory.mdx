use anyhow::Result;
use opendal::Operator;
use opendal::Scheme;
use std::collections::HashMap;

#[tokio::main]
async fn main() -> Result<()> {
    let mut map = HashMap::new();
    map.insert("root".to_string(), "/path/to/dir".to_string());

    let op: Operator = Operator::via_map(Scheme::Memory, map)?;
    Ok(())
}
